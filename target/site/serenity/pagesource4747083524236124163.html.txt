function cust_tuschdet_ONBLUR1(obj)
{
	if(mopId == "HOAACTU" || mopId == "hoaactu")
	{

		var frm = document.forms[0];

		if(obj.id == "cTenureDays")
		{
			if (!fnIsNull(obj.value)) 
			{
				var locNum = getValInStdFormat(obj.value);
				if ((isNaN(locNum)) || (locNum.indexOf(DEF_DECIMAL_SEPARATOR) != -1)) 
				{
					if (valSwitch)
					{
						switchCalArr[1] = finbranchResArr.get("FAT000201");
					}
					else
					{
						alert(finbranchResArr.get("FAT000201"));
						obj.focus();
						return false;
					}
				}
			}
			else
			{
				return true;
			}

			if(parseInt(frm.cTenureDays.value%30)!=0)
			{
				frm.custIPF.value=frm.cTenureDays.value;
				frm.custIPF.readOnly = true;
			}
			else
			{
				frm.custIPF.readOnly = false;
			}			

			if (!fnIsNull(frm.custAcctOpenDate.value))
			{
				 cust_populateMatDate();
			}


			if((frm.cIntRate.value!="") && (frm.cIntRate.value > 0))
			{
				 cust_populateRates();
			}
		}

		if(obj.id == "cIntRate")
		{
			if (fnIsNull(frm.cTenureDays.value) || (frm.cTenureDays.value < 0))
			{	
				frm.cIntRate.value = "";
				alert(customJspResArr.get("CTUA016508"));
				frm.cTenureDays.focus();
				return false;
			
			}		

			if (!fnIsNull(obj.value))
			{
				var locPcnt = getValInStdFormat(obj.value);
				var chk = (isNaN(locPcnt) || locPcnt.indexOf(" ") != -1 || parseFloat(locPcnt)>100) || (parseFloat(locPcnt) < 0);

				if (chk)
				{
					if (valSwitch)
					{
						switchCalArr[1] = finbranchResArr.get("FAT000167");
					}
					else
					{
						alert(finbranchResArr.get("FAT000167"));
						obj.focus();
						return false;
					}
				}        
			}
			
			if(!fnIsNull(obj.value))
			{
				 cust_populateRates();
		
			}


		}
	}
	return true;
}

		var scrName = "td_maturityDateCalc.scr";
		var retVal = fnExecuteScript(inputNameValues,outputNames,scrName,false);

			if(valueFromScript[0] == "custMaturityDate")

			if(valueFromScript[2] == "custDepPerdMths")

			{
